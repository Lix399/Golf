rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
  	
    function checkArgs(){
    	let args = ["time", "username", "levelNumber"];
      let timeArg = request.resource.data;
      
      return (
				timeArg.keys().hasOnly(args) &&    
        timeArg.keys().hasAll(args) &&
        timeArg.time is number && timeArg.time >= 0 &&
        timeArg.username is string && timeArg.username.size() > 0  &&
        timeArg.levelNumber is number && timeArg.levelNumber >= 0
      )
    }
  
    match /times/{id_times} {
    	allow delete, update: if false;
      allow create: if checkArgs();
      allow read: if true;
    }
  }
}
